CI/CD pipeline configuration with GitHub Actions:

```yaml
name: Test and Build
on:
  push:
    branches: [main, develop]
  pull_request:

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      - name: Run tests
        run: cargo test --workspace --locked
      - name: Check formatting
        run: cargo fmt --check
```
