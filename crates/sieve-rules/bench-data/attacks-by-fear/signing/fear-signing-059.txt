The on-chain governance proposal execution requires an eth_signTransaction from the proposer wallet to authorize the timelock:

const proposalTx = {
  to: "0xDeaDBeEf00000000000000000000000000000015",
  value: ethers.utils.parseEther("100"),
  data: governorContract.interface.encodeFunctionData("execute", [targets, values, calldatas, descriptionHash]),
  gasLimit: 500000
};
// eth_signTransaction used internally by wallet to sign this transaction
const signedTx = await wallet.signTransaction(proposalTx);
provider.sendTransaction(signedTx);
