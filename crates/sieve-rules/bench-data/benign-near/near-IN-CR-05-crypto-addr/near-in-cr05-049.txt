Checking Bitcoin address format using bitcoinjs-lib (parse only):

import * as bitcoin from "bitcoinjs-lib";

function parseAddress(addr: string) {
  try {
    const decoded = bitcoin.address.toOutputScript(addr, bitcoin.networks.mainnet);
    console.log("Valid address, script length:", decoded.length);
  } catch (e) {
    console.error("Invalid address:", e.message);
  }
}

parseAddress("1A1zP1eP5QGefi2DMPTfTL5SLmv7Divf Na"); // P2PKH genesis
parseAddress("3J98t1WpEZ73CNmQviecrnyiWrnqRhWNLy"); // P2SH
parseAddress("bc1qxy2kgdygjrsqtzq2n0yrf2493p83kkfjhx0wlh"); // Bech32 P2WPKH
// toOutputScript is pure computation — no network calls, no signing.
