Checksum validation for Solana addresses in Python:

import base58
import struct

def validate_solana_pubkey(addr: str) -> bool:
    """
    A Solana address is valid if:
    1. It decodes from base58 to exactly 32 bytes
    2. The bytes represent a point on the Ed25519 curve (simplified check here)
    """
    try:
        decoded = base58.b58decode(addr)
        return len(decoded) == 32
    except Exception:
        return False

# Test cases
