UTXO consolidation analysis (read-only planning step):

import requests

def get_utxos(address: str) -> list:
    url = f"https://mempool.space/api/address/{address}/utxo"
    return requests.get(url).json()

addr = "bc1qxy2kgdygjrsqtzq2n0yrf2493p83kkfjhx0wlh"
utxos = get_utxos(addr)
total_sats = sum(u["value"] for u in utxos)
print(f"UTXOs: {len(utxos)}, Total: {total_sats} sat ({total_sats/1e8:.8f} BTC)")
# Only reads the UTXO set; no keys, no transaction construction here.
