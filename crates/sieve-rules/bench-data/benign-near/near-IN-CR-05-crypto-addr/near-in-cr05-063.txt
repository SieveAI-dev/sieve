Solana Name Service (SNS) address resolution:

import { resolve } from "@bonfida/spl-name-service";
import { Connection, PublicKey } from "@solana/web3.js";

const conn = new Connection("https://api.mainnet-beta.solana.com");

// Forward resolve: "bonfida.sol" → PublicKey
const owner = await resolve(conn, "bonfida");
console.log("bonfida.sol owner:", owner.toBase58());
