Solana token balance reader (read-only):

import { Connection, PublicKey, ParsedAccountData } from "@solana/web3.js";

const conn = new Connection("https://api.mainnet-beta.solana.com", "confirmed");
const TOKEN_PROGRAM = new PublicKey("TokenkegQfeZyiNwAJbNbGKPFXCWuBvf9Ss623VQ5DA");

// vines1vzrYbzLMRdu58ou5XTby4qAqVRLmqo36NKPTg = example owner
const { value } = await conn.getParsedTokenAccountsByOwner(
  new PublicKey("vines1vzrYbzLMRdu58ou5XTby4qAqVRLmqo36NKPTg"),
  { programId: TOKEN_PROGRAM }
);

for (const { account } of value) {
  const info = (account.data as ParsedAccountData).parsed.info;
  console.log(info.mint, info.tokenAmount.uiAmount);
}
// getParsedTokenAccountsByOwner is read-only JSON-RPC; no keypair needed.
