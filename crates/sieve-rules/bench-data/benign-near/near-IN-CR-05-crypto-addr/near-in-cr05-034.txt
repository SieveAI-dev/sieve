# Solana address validation using @solana/web3.js

from solders.pubkey import Pubkey

def is_valid_solana_address(addr: str) -> bool:
    """Check if a string is a valid Solana public key (base58, 32 bytes)."""
    try:
        pk = Pubkey.from_string(addr)
        return True
    except Exception:
        return False

print(is_valid_solana_address("vines1vzrYbzLMRdu58ou5XTby4qAqVRLmqo36NKPTg"))  # True
print(is_valid_solana_address("not-a-valid-address"))  # False
print(is_valid_solana_address("9xQeWvG816bUx9EPjHmaT23yvVM2ZWbrrpZb9PusVFin"))  # True (Serum DEX)
