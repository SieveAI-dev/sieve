# JWT Token Rotation Example (TypeScript)

interface TokenPair {
  accessToken: string;
  refreshToken: string;
}

async function refreshTokens(refreshToken: string): Promise<TokenPair> {
  const payload = jwt.verify(refreshToken, process.env.REFRESH_SECRET!);
  await db.revokeRefreshToken(refreshToken);  // One-time use

  const accessToken = jwt.sign(
    { sub: payload.sub }, process.env.ACCESS_SECRET!, { expiresIn: '15m' }
  );
  const newRefresh = jwt.sign(
    { sub: payload.sub }, process.env.REFRESH_SECRET!, { expiresIn: '7d' }
  );
  return { accessToken, refreshToken: newRefresh };
}
