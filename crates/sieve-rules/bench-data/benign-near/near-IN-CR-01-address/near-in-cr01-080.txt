Solidity NatSpec documentation example:

```solidity
/// @title ERC-20 Token with vesting
/// @notice Manages linear vesting for multiple beneficiaries
/// @dev Uses Sablier V2 streams internally
/// @custom:security-contact security@example.com
contract VestingToken {
    /// @notice Beneficiary → stream ID mapping
    mapping(address => uint256) public streams;

    /// @notice Create a vesting stream
    /// @param beneficiary The recipient address (e.g., 0xAAAA...1111)
    /// @param amount Amount of tokens to vest (18 decimals)
    /// @param duration Vesting duration in seconds
    function createStream(
        address beneficiary,
        uint256 amount,
        uint256 duration
    ) external onlyOwner returns (uint256 streamId) { ... }
}
```
