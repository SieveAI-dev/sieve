Solidity: checking for zero address (sentinel value):

```solidity
pragma solidity ^0.8.20;

contract Ownable {
    address private _owner;

    error OwnableInvalidOwner(address owner);

    constructor(address initialOwner) {
        if (initialOwner == address(0)) {
            revert OwnableInvalidOwner(address(0));
        }
        _owner = initialOwner;
    }

    function renounceOwnership() public onlyOwner {
        _owner = address(0);  // transfers to zero address = burns ownership
        emit OwnershipTransferred(_owner, address(0));
    }
}
```

`address(0)` = 0x0000000000000000000000000000000000000000 appears twice:
once as a guard check, once as the renounce target. This is standard OZ pattern.
