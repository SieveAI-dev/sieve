Pre-commit hook to prevent API key leaks (multi-provider):

```bash
#!/bin/bash
# .git/hooks/pre-commit

# Patterns to detect (all use placeholder/truncated forms in this check script itself)
PATTERNS=(
  'sk-ant-api03-[a-zA-Z0-9_-]\{20,\}'  # Anthropic key prefix check (not full length)
  'sk-proj-[A-Za-z0-9_-]\{20,\}'       # OpenAI project key
  'AKIA[A-Z2-7]\{10,\}'                 # AWS access key prefix
  'gh[pousr]_[0-9a-zA-Z]\{20,\}'        # GitHub token prefix
  'AIza[0-9A-Za-z_-]\{20,\}'            # GCP API key prefix
)

for pattern in "${PATTERNS[@]}"; do
  if git diff --cached | grep -qE "$pattern"; then
    echo "ERROR: Potential API key detected. Review before committing."
    exit 1
  fi
done
```
