#!/bin/bash
# SSH key inventory audit script - lists public key fingerprints (no private key content)
set -euo pipefail

echo "=== SSH Key Inventory ==="
echo "Scanning ~/.ssh/ for key pairs..."

for pubkey in ~/.ssh/*.pub; do
    if [[ -f "$pubkey" ]]; then
        keyfile="${pubkey%.pub}"
        fingerprint=$(ssh-keygen -l -f "$pubkey" 2>/dev/null || echo "UNREADABLE")
        modified=$(stat -f "%Sm" -t "%Y-%m-%d" "$pubkey" 2>/dev/null || \
                   stat -c "%y" "$pubkey" 2>/dev/null | cut -d' ' -f1)
        
        echo "---"
        echo "Public key:  $pubkey"
        echo "Private key: $keyfile $(test -f "$keyfile" && echo "(exists)" || echo "(MISSING)")"
        echo "Fingerprint: $fingerprint"
        echo "Modified:    $modified"
    fi
done

echo ""
echo "=== authorized_keys ==="
if [[ -f ~/.ssh/authorized_keys ]]; then
    echo "Entry count: $(grep -c . ~/.ssh/authorized_keys)"
    ssh-keygen -l -f ~/.ssh/authorized_keys 2>/dev/null || echo "Could not parse"
fi
