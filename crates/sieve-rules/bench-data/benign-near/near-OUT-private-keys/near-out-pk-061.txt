PEM parsing errors across programming languages:

Python (cryptography library):
  from cryptography.exceptions import UnsupportedAlgorithm, ValueError
  try:
      key = load_pem_public_key(data)
  except ValueError as e:
      print(f"Invalid PEM data: {e}")
  # ValueError: Could not deserialize key data

Node.js:
  const { createPublicKey } = require('crypto');
  try {
      createPublicKey({ key: pemData, format: 'pem' });
  } catch (e) {
      console.error('PEM parse error:', e.code);
      // Error code: ERR_OSSL_PEM_NO_START_LINE
  }

Go (crypto/tls):
  block, _ := pem.Decode(pemData)
  if block == nil {
      log.Fatal("failed to decode PEM block containing certificate")
  }

Rust (rustls):
  let certs = rustls_pemfile::certs(&mut reader).collect::<Result<Vec<_>, _>>()?;
  // Error: io::Error with "no items found" if PEM unrecognized
