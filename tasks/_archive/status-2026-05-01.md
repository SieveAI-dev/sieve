## 四、关键命令速查

```bash
# 全量验证
cargo fmt --all -- --check
cargo clippy --workspace --all-targets --all-features --locked -- -D warnings
cargo test --workspace --locked
cargo deny check

# 带行为序列 feature 测试
cargo test --workspace --features sieve-cli/sequence_detection --locked

# benchmark + FP 率
cargo bench -p sieve-rules
cargo test -p sieve-rules --test dataset_fp_rate --release

# fuzz 60s 各 target
cargo +nightly fuzz run sse_parser -- -max_total_time=60
cargo +nightly fuzz run tool_use_aggregator -- -max_total_time=60
cargo +nightly fuzz run inbound_filter -- -max_total_time=60

# 启动透传 daemon
RUST_LOG=info cargo run -p sieve-cli -- start --config sieve.toml

# Dry-run 模式（仅记录命中，不拦截）
RUST_LOG=info cargo run -p sieve-cli -- start --config sieve.toml --dry-run

# 一键安装（dry-run）
cargo run -p sieve-cli -- setup --all-detected --dry-run

# 用户规则管理
cargo run -p sieve-cli -- rules list
cargo run -p sieve-cli -- rules edit
cargo run -p sieve-cli -- rules disable <rule-id>
cargo run -p sieve-cli -- rules enable <rule-id>

# 行为序列检测（opt-in）
cargo build --features sieve-cli/sequence_detection

# hot-reload 用户规则（daemon 运行时）
kill -SIGHUP $(pgrep sieve)
```
